CREATE TABLE moviesgross(
  moviegross_id INT GENERATED ALWAYS AS IDENTITY,
  year INT NOT NULL,
  rank INT NOT NULL,
  title VARCHAR(100) NOT NULL,
  release_date VARCHAR(30) NOT NULL,
  distributor VARCHAR(75) NOT NULL,
  genre VARCHAR(75) NOT NULL,
  gross INT NOT NULL,
  tickets_sold INT NOT NULL
);

CREATE TABLE movieinfo(

  released VARCHAR(25) NOT NULL,
  runtime VARCHAR(100) NOT NULL,
  genre VARCHAR(1000) NOT NULL,
  director VARCHAR(1000) NOT NULL,
  writer VARCHAR(1000) NOT NULL,
  actors VARCHAR(1000) NOT NULL,
  plot VARCHAR(20000) NOT NULL,
  language VARCHAR(2000) NOT NULL,

);
